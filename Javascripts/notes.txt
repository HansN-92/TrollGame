function hideButton(){
    if(buyButtonStrRing == true){
        document.getElementById("strRing").style.visibility = 'hidden';
    }
    else if(buyButtonStrRing == false){
        document.getElementById("strRing").style.visibility = 'visible';
    }

    if(buyButtonHPArmor == true){
        document.getElementById("maxHPArmor").style.visibility = 'hidden';
    }
    else if(buyButtonHPArmor == false){
        document.getElementById("maxHPArmor").style.visibility = 'visible';
    }

    if(buyButtonSpeedBoots == true){
        document.getElementById("speedBoots").style.visibility = 'hidden';
    }
    else if(buyButtonSpeedBoots == false){
        document.getElementById("speedBoots").style.visibility = 'visible';
    }
}
